<script>

/* ===== GLOBAL VARIABLE ===== */
let currentLocation = "Unknown";

/* ===== LOG FUNCTION =====
   Displays system messages in UI */
function log(msg) {
    document.getElementById("log").innerHTML += "<p>" + msg + "</p>";
}

/* ===== GPS LOCATION FETCH =====
   Gets user location using browser API */
navigator.geolocation.getCurrentPosition(pos => {
    currentLocation = `Lat: ${pos.coords.latitude}, Lon: ${pos.coords.longitude}`;
    document.getElementById("location").innerText = currentLocation;
});

/* ===== SEND DATA TO BACKEND =====
   Sends emergency details to server */
async function sendToServer(type) {
    try {
        let res = await fetch("http://localhost:3000/emergency", {
            method: "POST",
            headers: {"Content-Type": "application/json"},
            body: JSON.stringify({
                type: type,                  // Emergency type
                location: currentLocation,   // GPS data
                time: new Date()             // Timestamp
            })
        });

        let data = await res.json();

        /* Update AI severity result */
        document.getElementById("severity").innerText = 
            "Severity: " + data.severity;

        /* Show AI response */
        log("🤖 AI: " + data.message);

    } catch (e) {
        /* Offline fallback (LoRa simulation) */
        log("📡 No internet → Offline Mode (LoRa simulated)");
    }
}

/* ===== SOS FUNCTION =====
   Manual emergency trigger */
function triggerSOS() {
    document.getElementById("alertBox").style.display = "block";
    log("🚨 SOS Triggered!");

    /* Send data to backend */
    sendToServer("SOS");
}

/* ===== VOICE AI =====
   Detects 'help' voice command */
function startVoice() {
    const recognition = new webkitSpeechRecognition();

    recognition.onresult = function(event) {
        let speech = event.results[0][0].transcript;

        log("🎤 " + speech);

        /* If user says help → trigger SOS */
        if (speech.includes("help")) triggerSOS();
    };

    recognition.start();
}

/* ===== FALL DETECTION =====
   Simulates accident detection */
function simulateFall() {
    log("⚠️ Fall detected!");

    setTimeout(() => {
        log("🚨 No movement → SOS triggered");
        triggerSOS();
    }, 3000);
}
